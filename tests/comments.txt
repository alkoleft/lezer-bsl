# Комментарий в начале строки
// Это комментарий
Возврат 1;
==>
Module(
  LineComment,
  ReturnStmt(...))

# Комментарий в конце строки
Возврат 1; // Комментарий после кода
==>
Module(
  ReturnStmt(...),
  LineComment)

# Комментарий перед выражением
// Комментарий перед кодом
Возврат 1;
==>
Module(
  LineComment,
  ReturnStmt(...))

# Комментарий между выражениями
Возврат 1;
// Комментарий между выражениями
Возврат 2;
==>
Module(
  ReturnStmt(...),
  LineComment,
  ReturnStmt(...))

# Комментарий внутри процедуры
Процедура Тест()
  // Комментарий внутри процедуры
  а = 1;
  // Комментарий внутри процедуры
  а = 1;
КонецПроцедуры
==>
Module(ProcedureDef(Name,ParamList,
  LineComment,
  Block(
    AssignmentStmt(...),
    LineComment,
    AssignmentStmt(...))))

# Комментарий перед возвратом
Функция Сумма()
  // Комментарий перед возвратом
  Возврат 1;
КонецФункции
==>
Module(FunctionDef(Name,ParamList,
  LineComment,
  Block(...)))

# Множественные комментарии
// Первый комментарий
// Второй комментарий
Возврат 1;
// Третий комментарий
Возврат 2;
==>
Module(
  LineComment,
  LineComment,
  ReturnStmt(...),
  LineComment,
  ReturnStmt(...))

# Комментарий перед условным оператором
Если Истина Тогда
  // Комментарий в блоке
  а = 1;
КонецЕсли;
==>
Module(
  IfStmt(
    Condition(...),
    LineComment,
    Block(...)))

# Комментарий с выражением, содержащим деление
Возврат 1 / 2; // Комментарий после деления
==>
Module(
  ReturnStmt(...),
  LineComment)

# Пустой комментарий
// 
Возврат 1;
==>
Module(
  LineComment,
  ReturnStmt(...))
