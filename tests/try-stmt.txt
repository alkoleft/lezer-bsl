# Простой блок обработки исключений
Попытка
Исключение
КонецПопытки;
==>
Module(
  TryStmt(Block(),Block()))

# Блок обработки исключений с операторами
Попытка
  а = 1;
Исключение
  а = 0;
КонецПопытки;
==>
Module(TryStmt(
  Block(AssignmentStmt(...)),
  Block(...)))

# Блок обработки исключений с несколькими операторами в блоке Попытка
Попытка
  а = 1;
  б = 2;
  в = 3;
Исключение
  а = 0;
КонецПопытки;
==>
Module(TryStmt(
  Block(
    AssignmentStmt(...),
    AssignmentStmt(...),
    AssignmentStmt(...)),
  Block(AssignmentStmt(...))))

# Блок обработки исключений с несколькими операторами в блоке Исключение
Попытка
  а = 1;
Исключение
  а = 0;
  б = -1;
КонецПопытки;
==>
Module(TryStmt(
  Block(AssignmentStmt(...)),
  Block(
    AssignmentStmt(...),
    AssignmentStmt(...))))

# Блок обработки исключений на английском
Try
  a = 1;
Except
  a = 0;
EndTry;
==>
Module(TryStmt(
  Block(AssignmentStmt(...)),
  Block(AssignmentStmt(...))))

# Вложенный блок обработки исключений
Попытка
  Попытка
    а = 1;
  Исключение
    а = 0;
  КонецПопытки;
Исключение
  а = -1;
КонецПопытки;
==>
Module(TryStmt(
  Block(
    TryStmt(
      Block(AssignmentStmt(...)),
      Block(AssignmentStmt(...)))),
  Block(AssignmentStmt(...))))

# Блок обработки исключений с вызовом исключения
Попытка
  ВызватьИсключение "Ошибка";
Исключение
  Сообщить("Ошибка обработана");
КонецПопытки;
==>
Module(TryStmt(
  Block(RaiseStmt(...)),
  Block(CallStmt(...))))

# Блок обработки исключений с условным оператором
Попытка
  Если условие Тогда
    а = 1;
  КонецЕсли;
Исключение
  а = 0;
КонецПопытки;
==>
Module(TryStmt(
  Block(IfStmt(
    Condition(...),
    Block(...))),
  Block(AssignmentStmt(...))))

# Блок обработки исключений с циклом
Попытка
  Для а = 1 По 10 Цикл
    результат = результат + а;
  КонецЦикла;
Исключение
  результат = 0;
КонецПопытки;
==>
Module(TryStmt(
  Block(ForStmt(...)),
  Block(AssignmentStmt(...))))

# Блок обработки исключений с возвратом в блоке Попытка
Попытка
  Если условие Тогда
    Возврат 1;
  КонецЕсли;
Исключение
  Возврат 0;
КонецПопытки;
==>
Module(TryStmt(
  Block(IfStmt(...)),
  Block(ReturnStmt(...))))
