Module(RaiseStmt(Expression(...)))

# Оператор ВызватьИсключение в условном операторе
Если условие Тогда
  ВызватьИсключение "Ошибка";
КонецЕсли;
==>
Module(IfStmt(
  Condition(...),
  Block(RaiseStmt(Expression(...)))))

# Оператор ВызватьИсключение в блоке обработки исключений

Исключение

КонецПопытки;
==>
Module(TryStmt(
  Block(RaiseStmt(Expression(...))),
  Block(...)))

# Оператор ВызватьИсключение в цикле

==>
Module(WhileStmt(
  Condition(...),
  Block(RaiseStmt(Expression(...)))))

# Оператор ВызватьИсключение с созданием объекта

Module(RaiseStmt(Expression(...)))

# Оператор ВызватьИсключение внутри процедуры
Процедура Тест()
  ВызватьИсключение "Ошибка";
КонецПроцедуры
==>
Module(ProcedureDef(Name,ParamList,Block(
RaiseStmt(Expression(...)))))

# Оператор ВызватьИсключение внутри функции

  Возврат 1;
КонецФункции
==>

RaiseStmt(Expression(...)),
ReturnStmt(...))))
