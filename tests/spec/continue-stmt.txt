# Простой оператор Продолжить
Продолжить;
==>
Module(ModuleBlock(ContinueStmt))

# Простой оператор Continue
Continue;
==>
Module(ModuleBlock(ContinueStmt))

# Оператор Продолжить в цикле пока
Пока Истина Цикл
  Продолжить;
КонецЦикла;
==>
Module(ModuleBlock(WhileStmt(
  Condition(...),
  Block(ContinueStmt))))

# Оператор Продолжить в цикле для
Для а = 1 По 10 Цикл
  Продолжить;
КонецЦикла;
==>
Module(ModuleBlock(ForStmt(
  VariableName,
  Start(...),
  End(...),
  Block(ContinueStmt))))

# Оператор Продолжить в цикле для каждого
Для Каждого элемент Из коллекция Цикл
  Продолжить;
КонецЦикла;
==>
Module(ModuleBlock(ForEachStmt(
  VariableName,
  Collection(...),
  Block(ContinueStmt))))

# Оператор Продолжить с условием
Пока Истина Цикл
  Если условие Тогда
    Продолжить;
  КонецЕсли;
КонецЦикла;
==>
Module(ModuleBlock(WhileStmt(
  Condition(...),
  Block(IfStmt(
    Condition(...),
    Block(ContinueStmt))))))

# Оператор Продолжить во вложенном цикле
Для а = 1 По 10 Цикл
  Для б = 1 По 10 Цикл
    Продолжить;
  КонецЦикла;
КонецЦикла;
==>
Module(ModuleBlock(ForStmt(
  VariableName,
  Start(...),
  End(...),
  Block(ForStmt(
    VariableName,
    Start(...),
    End(...),
    Block(ContinueStmt))))))

# Оператор Продолжить с операторами до него
Пока Истина Цикл
  а = 1;
  Продолжить;
  б = 2;
КонецЦикла;
==>
Module(ModuleBlock(WhileStmt(
  Condition(...),
  Block(
    AssignmentStmt(...),
    ContinueStmt,
    AssignmentStmt(...)))))

# Оператор Продолжить в блоке обработки исключений
Пока Истина Цикл
  Попытка
    Продолжить;
  Исключение
    Сообщить("Ошибка");
  КонецПопытки;
КонецЦикла;
==>
Module(ModuleBlock(WhileStmt(
  Condition(...),
  Block(TryStmt(
    Block(ContinueStmt),
    Block(...))))))
