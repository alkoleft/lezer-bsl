  ReturnStmt))

# Оператор Возврат внутри функции
Функция Сумма()
  Возврат 1 + 2;
КонецФункции
==>
Module(FunctionDef(Name,ParamList,Block(
  ReturnStmt(Expression(...)))))

# Оператор Возврат в условном операторе
Если условие Тогда
  Возврат 1;
Иначе
  Возврат 0;
КонецЕсли;
==>
Module(ModuleBlock(IfStmt(
  Condition(...),
  Block(
    ReturnStmt(Expression(...))),
  ElseClause(Block(
    ReturnStmt(Expression(...)))))))

# Оператор Возврат в цикле
Пока Истина Цикл
  Возврат 1;
КонецЦикла;
==>
Module(ModuleBlock(WhileStmt(
  Condition(...),
  Block(ReturnStmt(Expression(...))))))
