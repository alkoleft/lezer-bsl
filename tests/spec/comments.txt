# Комментарий в начале строки
// Это комментарий
Возврат 1;
==>
Module(ModuleBlock(ReturnStmt(Expression(Number))))

# Комментарий в конце строки
Возврат 1; // Комментарий после кода
==>
Module(ModuleBlock(ReturnStmt(Expression(Number))))

# Комментарий между выражениями
Возврат 1;
// Комментарий между выражениями
Возврат 2;
==>
Module(ModuleBlock(
  ReturnStmt(Expression(Number)),
  ReturnStmt(Expression(Number))))

# Комментарий внутри процедуры
Процедура Тест()
  // Комментарий внутри процедуры
  а = 1;
  // Комментарий внутри процедуры
  а = 1;
КонецПроцедуры
==>
Module(ProcedureDef(Name,ParamList,
  Block(
    AssignmentStmt(VariableName, Expression(Number)),
    AssignmentStmt(VariableName, Expression(Number)))))

# Комментарий перед возвратом
Функция Сумма()
  // Комментарий перед возвратом
  Возврат 1;
КонецФункции
==>
Module(FunctionDef(Name,ParamList,
  Block(ReturnStmt(Expression(Number)))))

# Множественные комментарии
// Первый комментарий
// Второй комментарий
Возврат 1;
// Третий комментарий
Возврат 2;
==>
Module(ModuleBlock(
  ReturnStmt(Expression(Number)),
  ReturnStmt(Expression(Number))))

# Комментарий перед условным оператором
Если Истина Тогда
  // Комментарий в блоке
  а = 1;
КонецЕсли;
==>
Module(ModuleBlock(
  IfStmt(
    Condition(True),
    Block(AssignmentStmt(VariableName, Expression(Number))))))

# Комментарий с выражением, содержащим деление
Возврат 1 / 2; // Комментарий после деления
==>
Module(ModuleBlock(
  ReturnStmt(Expression(BinaryExpr(Number, ArithOp, Number)))))

# Пустой комментарий
// 
Возврат 1;
==>
Module(ModuleBlock(ReturnStmt(Expression(Number))))
